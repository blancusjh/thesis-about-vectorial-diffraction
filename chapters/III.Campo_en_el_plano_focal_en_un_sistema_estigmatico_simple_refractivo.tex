\chapter{Campo en el plano focal en un sistema estigmático simple refractivo}


En la sección presente, se cálcula el campo en el plano focal en un sistema estigmático simple (formado por una sola superficie dioptrica). Asumimos que incide un campo esférico desde uno de los puntos estigmáticos $A$ y calcularemos el campo refractado en un plano perpendicular al eje óptico que contenga a el otro punto estigmático $A'$.

Partimos primero, por escribir las cantidades geometricas relevantes de las superficies cartesianas para nuestro problema: 

Las ecuaciónes parametricas de las superficies dioptricas 

% TODO: agrupar z y r en un entorno Align.
\begin{equation}
  z(\rho) = \frac{(O + T\rho^2)\rho^2}{1 + S\rho^2 + \sqrt{1 + (2S - O^2G)\rho^2}},
  \label{eq:parametrical_ovoid_z_app}
\end{equation}




\begin{equation}
  r(\rho) = \sqrt{\rho^2 - z^2(\rho)}.
  \label{eq:parametrical_ovoid_r_app}
\end{equation}

El vecotor normal unitario en cada punto de la superficie considerada

\begin{equation}
  \mathbf{\hat N}(\rho,\varphi)=
  \frac{r'(\rho)\,\mathbf{\hat z}-z'(\rho)\,\mathbf{\hat r}(\varphi)}
  {\sqrt{(r'(\rho))^2+(z'(\rho))^2}}.
  \label{eq:N_ovoid_rho_phi_app}
\end{equation} %TODO: Citarla de la ya enteriormente escrita en † óptica geométrica. (eq:N_ovoid_rho_phi)


Los vectores unitarios de incidencia y transmisión del sistema 

\begin{equation}
  \uvec{u} = \frac{(z(\rho) - z_0)\uvec{z} + r(\rho)\uvec{r}(\varphi)}{\sqrt{\rho^2 - 2z(\rho)z_0 + z_0^2}},
  \label{eq:u_for_P_in_optical_axis_app}
\end{equation}

\begin{equation}
  \uvec{u}' = \frac{(z(\rho) - z_i)\uvec{z} + r(\rho)\uvec{r}(\varphi)}{\sqrt{\rho^2 - 2z(\rho)z_i + z_i^2}}.
  \label{eq:u_prime_for_P_in_optical_axis_app}
\end{equation}


De la teoria electromagnética, sabemos que para sistemas axialmente simetricos el campo sobre la interfaz $\Sigma$ que se transmite, se calcula como 




\begin{equation}
  \vb{E}'_{\Sigma} = t_p (\vb{E}_\Sigma \cdot \uvec{p}) \uvec{p}' + t_s (\vb{E}_\Sigma \cdot \uvec s) \uvec{s}, 
  \label{eq:Transmited_E_field_in_Sigma_app}
\end{equation}

donde $\uvec s$, $\uvec p$ y $\uvec p'$ se definen como

\begin{equation}
    \uvec s = \frac{\uvec N \times \uvec u}{\lvert  \uvec N \times \uvec u \rvert},
    \label{eq:s_hat_def_app}
\end{equation}


\begin{equation}
    \uvec p = \uvec s \times \uvec u   
    \label{eq:p_hat_def_app}.
\end{equation}

\begin{equation}
    \uvec p' = \uvec s \times \uvec u'  
    \label{eq:p_prime_hat_def}.
\end{equation}


que remplazando por las ecuaciones \ref{eq:N_ovoid_rho_phi_app}, \ref{eq:u_for_P_in_optical_axis} y \ref{eq:u_prime_for_P_in_optical_axis} 

\begin{equation}
  \uvec p = \frac{1}{\sqrt{\rho^2 - 2zz_0 + z_0^2}} \left\{(z(\rho) - z_0)\uvec{r}(\phi) - r(\rho)\uvec{z} \right\} 
  \label{eq:p_mode_ovoid}
\end{equation}


\begin{equation}
  \uvec{p}' = \frac{1}{\sqrt{\rho^2 - 2zz_i + z_i^2}} \left\{(z(\rho) - z_i)\uvec{r}(\phi) - r(\rho)\uvec{z} \right\} 
  \label{eq:p_prime_mode_ovoid}
\end{equation}

\begin{equation}
  \uvec{s} = \uvec{\phi}(\phi)
  \label{eq:s_mode_ovoid}
\end{equation}



Y los coeficientes de Fresnel de transmición, se definen en funcion de los cosenos de los ángulos de incidencia y de refracción como:  


\begin{equation}
t_s = \frac{2 n_1 \cos \theta_i}{n_1 \cos \theta_i + n_2 \cos \theta_t}
\label{eq:ts_app}
\end{equation}


\begin{equation}
t_p = \frac{2 n_1 \cos \theta_i}{n_2 \cos \theta_i + n_1 \cos \theta_t}
\label{eq:tp_app}
\end{equation}


Los cosenos $\cos \theta_i$  y $\cos \theta_t$ los obtenemos a partir de las siguientes expresiones 


\begin{align}
  \cos\theta_i &= -\uvec{N} \cdot \uvec{u} \\
  \cos\theta_t &= -\uvec{N} \cdot \uvec{u}'
\end{align}


% ----------------


\begin{equation}
  \uvec N(\rho,\varphi)=
  \frac{r'(\rho)\,\uvec z-z'(\rho)\,\uvec r(\varphi)}
  {\sqrt{(r'(\rho))^2+(z'(\rho))^2}}
\label{eq:N_ovoid_rho_phi_explicit}
\end{equation}


\begin{align}
\cos\theta_i(\rho)
&= -\,\uvec N(\rho,\varphi)\cdot\uvec u(\rho,\varphi)
\label{eq:cos_theta_i_def_app}
\\[2mm]
&= -\;
\frac{
\big(r'(\rho)\uvec z-z'(\rho)\uvec r(\varphi)\big)\cdot\big((z(\rho)-z_i)\uvec z+r(\rho)\uvec r(\varphi)\big)
}{
\sqrt{(r'(\rho))^2+(z'(\rho))^2}\;
\sqrt{\rho^2-2z(\rho)z_i+z_i^2}
}
\label{eq:cos_theta_i_closed_app}
\end{align}

\begin{equation}
  \cos\theta_i(\rho) = \frac{
z'(\rho)\,r(\rho)\;-\;r'(\rho)\,\big(z(\rho)-z_i\big)
}{
\sqrt{(r'(\rho))^2+(z'(\rho))^2}\;
\sqrt{\rho^2-2z(\rho)z_i+z_i^2}
}.
  \label{eq:cos_theta_i_ovoid_z_and_r}
\end{equation}


\begin{equation}
\cos\theta_i(\rho)
=
\frac{
z'(\rho)\,\big(\rho^2-z(\rho)z_0\big)-\rho\big(z(\rho)-z_0\big)
}
{
\sqrt{\rho^2\big(1+z'(\rho)^2\big)-2\rho z(\rho)z'(\rho)}\;
\sqrt{\rho^2-2z(\rho)z_0+z_0^2}
}.
\label{eq:cos_theta_i_ovoid}
\end{equation}


Luego, $cos\theta_t$ se obtiene simplemente remplazando $z_0$ por $z_i$: 

\begin{equation}
\cos\theta_t(\rho)
=
\frac{
z'(\rho)\,\big(\rho^2-z(\rho)z_i\big)-\rho\big(z(\rho)-z_i\big)
}
{
\sqrt{\rho^2\big(1+z'(\rho)^2\big)-2\rho z(\rho)z'(\rho)}\;
\sqrt{\rho^2-2z(\rho)z_i+z_i^2}
}.
\label{eq:cos_theta_t_ovoid}
\end{equation}


Así pues, los cosenos de incidencia y transmisión son


\begin{equation}
\left\{
\begin{aligned}
\cos\theta_i &=
\frac{
z'\,\big(\rho^2 - z\,z_0\big) - \rho\big(z - z_0\big)
}
{
\sqrt{\rho^2\big(1+z'^2\big) - 2\rho z z'}\;
\sqrt{\rho^2 - 2 z z_0 + z_0^2}
}
\\[8pt]
\cos\theta_t &=
\frac{
z'\,\big(\rho^2 - z\,z_i\big) - \rho\big(z - z_i\big)
}
{
\sqrt{\rho^2\big(1+z'^2\big) - 2\rho z z'}\;
\sqrt{\rho^2 - 2 z z_i + z_i^2}
}
\end{aligned}
\right.
\label{cosines_ovoid_compact}
\end{equation}


Usando las definiciones 

\begin{align}
l_0 &= \sqrt{\rho^2 - 2zz_0 + z_0^2} \\ 
l_i &= \sqrt{\rho^2 - 2zz_i + z_i^2} 
\label{eq:l0_li_definitions}
\end{align}

que son las distancias $\overline{AQ}$ y $\overline{A'Q}$ respectivamente. 

y definiendo tambien 

\begin{equation}
  s' := \frac{ds}{d\rho} = \sqrt{(z')^2 + (r')^2},
  \label{eq:ds_drho}
\end{equation}


los cosenos de incidencia y transmisión quedan expresados como: 


\begin{equation}
\begin{aligned}
  \cos\theta_i &= \frac{1}{s'l_0}\big\{z'\big(\rho^2 - z z_0\big) - \rho\big(z - z_0\big)
\big\} \\[8pt]
 \cos\theta_t &= \frac{1}{s'l_i}\big\{z'\big(\rho^2 - z z_i\big) - \rho\big(z - z_i\big)
\big\} \\[8pt]
\end{aligned}
\label{cosines_ovoid}
\end{equation}

Remplazando \ref{cosines_ovoid_compact} en \ref{eq:ts_app} y en \ref{eq:tp_app}, obtene os los coeficientes de fresnel para la transmisión para la superficie cartesiana parametrizada por los parametros $G, O, T, S$ 


\begin{equation}
 t_s = \frac{2 \left\{ z'(\rho^2 - zz_0) - \rho(z - z_0) \right\}}{ \left\{ z'(\rho^2 - zz_0) - \rho(z - z_0) \right\} + \eta \frac{l_0}{l_i} \left\{ z'(\rho^2 - zz_i) - \rho(z - z_i) \right\} },
  \label{eq:t_s_ovoid}
\end{equation}


\begin{equation}
 t_p = \frac{2 \left\{ z'(\rho^2 - zz_0) - \rho(z - z_0) \right\}}{ \eta \left\{ z'(\rho^2 - zz_0) - \rho(z - z_0) \right\} + \frac{l_0}{l_i} \left\{ z'(\rho^2 - zz_i) - \rho(z - z_i) \right\} }.
  \label{eq:t_p_ovoid}
\end{equation}


Con esto, terminamos de proporcionar la forma exacta de las cantidades relevantes necesarias 
para el cálculo del campo eleçtrico en el plano focal en función de la función sagita que 
define el dioptrio estigmático de interés.



\section{Campo Eléctrico en el plano Focal}

El problema que se resolverá en la sección presente, es la de determinar el campo eléctrico en el plano focal dad oun campo inicial cualquiera, el cual posteriormente especificaremos como el campo incididente correspondiente a una fuente puntual centrada en uno de los puntos estigmáticos, pero en principio los calculos aqui presentados son de una naturaleza general.

\textit{Denotaremos como $\vb E$ en general, al campo incidente, y a $\vb E'$ el campo tras la refracción.}


Sea $\vb{E}_p(\vb{r})$ el campo eléctrico en el plano $P$ $\{\vb{E}_p' := \vb{E}(P)\}$.
Calcúlese el campo en $P'$ a una distancia $d$ del plano $P$.

%NOTE : Figura que ilustre el setup.


Transfiérase $\vb{E}_p$ a la superficie $\Sigma$ adicionando la fase que acumula el campo en su trayecto.
Considerar que el campo describe una línea recta en su recorrido de $p$ a $q$ y asumir que dicho recorrido es aproximadamente horizontal, justificando esto porque consideraremos como primera aproximación un ángulo $\alpha$ moderado.

Con tales consideraciones, el campo $\vb{E}_{\Sigma}$, sobre $\Sigma$ estará dado por

\begin{equation}
    \vb{E}_{\Sigma}(\vb{r}) = e^{ikz(\vb{r})} \vb{E}_p(\vb{r})
    \label{eq:Ep_from_E_Sigma}
\end{equation}

\begin{equation}
    \vb{E}_{\Sigma}(r,\phi,z(r)) = e^{ikz(r)} \vb{E}_p(r,\phi,z=0)
\end{equation}

Así, calculamos el campo transmitido sobre la interfaz $\Sigma$

%NOTE : Figura que ilustre el setup.

\begin{equation}
    \vb{E}'_{\Sigma}(\vb{q}) = t_p (\vb{E}_{\Sigma} \cdot \uvec{p}) \uvec{p}' + t_s (\vb{E}_{\Sigma} \cdot \uvec{s}) \uvec{s}
\label{eq:E_Sigma_Prime_in_terms_of_E_Sigma}
\end{equation}

Descomponiendo el campo en la interfaz en sus componentes cilíndricas:

\begin{equation}
    \vb{E}_{\Sigma} = E_r \uvec{r} + E_{\phi} \uvec{\phi} + E_z \uvec{z}
\end{equation}

Calculamos las proyecciones sobre los vectores de la base de Fresnel $\uvec{p}$ y $\uvec{\phi}$ (donde $\uvec{s} = \uvec{\phi}$):

\begin{equation}
    \vb{E}_{\Sigma} \cdot \uvec{p} = \frac{1}{\sqrt{\rho^2 - 2zz_0 + z_0^2}} \big\{ (z(\rho) - z_0) E_r - r(\rho) E_z \big\}
\end{equation}

\begin{equation}
    \vb{E}_{\Sigma} \cdot \uvec{\phi} = E_{\phi}
\end{equation}

Sustituyendo en la expresión del campo transmitido:

\begin{equation}
    \vb{E}'_{\Sigma} = t_p \frac{ \big\{ (z(\rho) - z_0) E_r - r(\rho) E_z \big\} }{\sqrt{\rho^2 - 2zz_0 + z_0^2} \sqrt{\rho^2 - 2zz_i + z_i^2}} \big\{ (z(\rho) - z_i) \uvec{r} - r(\rho) \uvec{z} \big\} + t_s E_{\phi} \uvec{\phi}
\end{equation}

Reemplazando los coeficientes de Fresnel explícitamente:

\begin{align}
    \vb{E}'_{\Sigma} &= 2 \frac{ \{ r'(z_i-z) + r z' \} }{ \{ r'(z_0-z) + r z' \} + \eta \{ r'(z_i-z) + r z' \} \frac{l_0}{l_i} } E_{\phi} \uvec{\phi} \nonumber \\[8pt]
    &\quad + 2 \frac{ \{ r'(z_i-z) + r z' \} }{ \eta \{ r'(z_0-z) + r z' \} + \{ r'(z_i-z) + r z' \} \frac{l_i}{l_0} } \big\{ (z(\rho) - z_0) E_r - r(\rho) E_z \big\} \uvec{p}'
\end{align}

Transfiérase el campo de vuelta a $P$ para hacer uso de los métodos de propagación, llamamos a este ``campo virtual transmitido'' en $P$ como $\vb{E}_P'$.

\begin{equation}
    \vb{E}_p' = e^{-inkz(r)} \vb{E}_{\Sigma}'.
\end{equation}

Remplazando según \eqref{eq:E_Sigma_Prime_in_terms_of_E_Sigma}



\begin{align}
    \vb{E}_p' &= e^{-inkz(r)} \big\{ t_p (\vb{E}_{\Sigma} \cdot \uvec{p}) \uvec{p}' + t_s (\vb{E}_{\Sigma} \cdot \uvec{\phi}) \uvec{\phi} \big\} \nonumber \\[6pt]
    &= e^{-inkz(r)} \big\{ t_p (e^{ikz(r)} \vb{E}_p \cdot \uvec{p}) \uvec{p}' + t_s (e^{ikz(r)} \vb{E}_p \cdot \uvec{\phi}) \uvec{\phi} \big\}
\end{align}

\begin{equation}
    \vb{E}_p' = e^{ik(1-n)z(r)} \big\{ t_p (\vb{E}_p \cdot \uvec{p}) \uvec{p}' + t_s (\vb{E}_p \cdot \uvec{\phi}) \uvec{\phi} \big\}
\end{equation}




Definimos el campo inicial en el plano $z=0$:

\begin{equation}
    \vb{E}(\vb r, z=0) := \vb{E}_p(\vb r)
\end{equation}

Y lo propagamos hasta el plano focal $z=f$:

\begin{equation}
    \vb{E}'(r, z=f) = \mathscr{R}_n [f] \big\{ \vb{E}_p'(r) \big\},
\end{equation}

donde $\mathscr R_n[f]$ denota el operador de propagación que toma un campo en un plano dado y da el campo en otro plano paralelo y a una distancia $f$ al primero en un medio de indice de refracción $n$.

Tomando como válida en nuestro estudio la aproximación de Fresnel, que hemos usado implicitamente cuando transferimos los campos del plano al dioptio y viceversa, tenemos que el operador de propagación esta dado por 


\begin{equation}
    \mathscr{R}_n [f] \{ U(x,y) \} =
    n\frac{e^{inkf}}{i \lambda f}
    \iint_{-\infty}^{\infty}
    U(x',y')
    \exp\!\left[
        \frac{ink}{2f}
        \lVert \vb{r}-\vb{r}' \rVert^2
    \right]
    d\vb{r}'
\end{equation}

Por lo tanto, el campo eléctrico en el plano focal esta dado por  

\begin{equation}
    \vb{E}'(r, z=f) =
    n\frac{e^{inkf}}{i \lambda f}
    \iint_{-\infty}^{\infty}
    \vb{E}_p'(r')
    \exp\!\left[
        \frac{ink}{2f}
        \lVert \vb{r}-\vb{r}' \rVert^2
    \right]
    d\vb{r}',
\end{equation}

o equivalentemente, 


\begin{equation}
    \vb{E}'(r, z) =
    n\frac{e^{inkf}}{i \lambda z}
    \iint_{-\infty}^{\infty}
    \exp\!\left[
        \frac{ink}{2z}
        \lVert \vb{r}-\vb{r}' \rVert^2
        + ik(1-n)z(r')
    \right]
    \Big\{
        t_p (\vb{E}_P \cdot \uvec{p}) \uvec{p}'
        +
        t_s (\vb{E}_P \cdot \uvec{\phi}) \uvec{\phi}
    \Big\}
    dx'dy'.
\label{eq:E_field_at_z_3D}
\end{equation}


Esta expresión nos da el campo en el plano paralelo a $P$ a una distancia $z$, dado el campo inicial en $P$. 


Este resultado puede ser tambien expresado en términos de una transformada de Fourier expandiendo el kernel de Fresnel %NOTE : CITE Goodman Fourier Optics  

\begin{equation}
\mathbf{E}'(\mathbf r,f)
=
n\frac{e^{inkf}}{i\lambda f}\,
\exp\!\left[
\frac{ink}{2f}\|\mathbf r\|^2
\right]
\,
\mathcal F\{\mathbf P\}
\left(
\frac{n\mathbf r}{\lambda f}
\right).
\label{E_in_focal_plane_as_FT_3D} 
\end{equation}

Donde, 

\begin{equation}
\mathbf P(\|\mathbf r'\|)
=
\exp\!\left[
\frac{ink}{2f}\|\mathbf r'\|^2
+
ik(1-n)z(\|\mathbf r'\|)
\right]
\left\{
t_p (\mathbf{E}_P\cdot \hat{\boldsymbol p}) \hat{\boldsymbol p}'
+
t_s (\mathbf{E}_P\cdot \hat{\boldsymbol\phi}) \hat{\boldsymbol\phi}
\right\},
\label{eq:P_definition}
\end{equation}


y 

\begin{equation}
\mathcal F\{\mathbf P\}(\boldsymbol{\nu})
=
\iint_{-\infty}^{\infty}
\mathbf P(\mathbf r')\,
e^{-i2\pi\,\boldsymbol{\nu}\cdot\mathbf r'}\,
dx'\,dy',
\qquad
\boldsymbol{\nu}=\frac{n\mathbf r}{\lambda f}.
\end{equation}

Defínase un sistema coordenado esférico centrado en el foco del sistema, que es el punto $A'$ , llámese por $\uvec R_{A'}, \uvec \theta_{A'}, \uvec \phi$ los vectores curvilíneos asociados al sistema. 

Los estados de polarización $\uvec p'$  y $\uvec s$, son la polarización definida por los vectores base del sistema esférico centrado en $A'$ tangentes a la esféra, esto es $\uvec \theta_{A'}$ y $\uvec \phi$ respectivamente, como puede ser comprabado con facilidad desde las expresiones \eqref{eq:p_prime_mode_ovoid} y \eqref{eq:s_mode_ovoid}.

De modo que en el sistema esférico centrado en $A'$ la base de polarización queda escrita como 

\begin{align}
\uvec{s} &= \uvec{\phi}
= -\sin\phi\,\uvec{x} + \cos\phi\,\uvec{y}, \\[6pt]
\uvec{p}' &= \uvec{\theta}_{A'}
= \cos\theta_{A'}(\cos\phi\,\uvec{x} +
\sin\phi\,\uvec{y}) -
\sin\theta_{A'}\,\uvec{z}.
\label{s_p_in_A_prime_system}
\end{align}

Si bien esta base sugiere un campo tridimensional, el problema puede restringirse al plano perpendicular al eje óptico mediante la condición de transversalidad, establece una dependencia entre el campo tranversal y longitudinal, permitiendo determinar el segundo conociendo el primero.

Escribiendo la condición de transversalidad, como que el campo ha de ser tangente al vector radial unitario con origen en $A'$ 

\begin{equation}
\vb{E}\cdot\uvec{R}_{A'}=0,
\label{eq:transversality_condition}
\end{equation}

lo que implica

\begin{equation}
E_r\sin\theta_{A'} + E_z\cos\theta_{A'} = 0,
\end{equation}

y por tanto

\begin{equation}
E_z = -\tan\theta_{A'}\,E_r.
\label{eq:Ez_from_Er}
\end{equation}

que es equivalente a


\begin{equation}
  E_z = -\frac{r}{z_i} \,E_r.
\label{eq:Ez_from_Er_rz}
\end{equation}

De esta manera, el campo transversal determina completamente el campo longitudinal, 
y por ende el campo vectorial $\vb E$ completamente.


Pasamos pues, sin perdida de generalridad al problema tranversal, denotamos $\vb E'_\perp$ como la parte transversal de $\vb E'$. 

De \eqref{eq:E_Sigma_Prime_in_terms_of_E_Sigma} se sigue que 



\begin{equation}
\mathbf{E}'_\perp(\mathbf r,f)
=
n\frac{e^{inkf}}{i\lambda f}\,
\exp\!\left[
\frac{ink}{2f}\|\mathbf r\|^2
\right]
\,
\mathcal F\{\mathbf P_\perp\}
\left(
\frac{n\mathbf r}{\lambda f}
\right).
\label{eq:E_field_as_FT}
\end{equation}

donde 

\begin{equation}
\mathbf P_\perp(\|\mathbf r'\|)
=
\exp\!\left[
\frac{ink}{2f}\|\mathbf r'\|^2
+
ik(1-n)z(\|\mathbf r'\|)
\right]
\left\{
t_p \mathbf{E}_{P,p} \uvec p'_\perp
+
t_s \mathbf{E}_{P,\phi} \uvec \phi
\right\}.
\end{equation}


y $\uvec p'_\perp$ y $\uvec s$ denotan las proyecciones transversales de las polarizaciones $\uvec p'$ y $\uvec s$ respectivamente.



\subsection{Resultado en la base cilindrica} 


Los vectores $\uvec s_\perp$ y $\uvec p_\perp$ estan dados por



\begin{align}
\uvec{s}_\perp &= \uvec{\phi}
= -\sin\phi\,\uvec{x} + \cos\phi\,\uvec{y}, \\
\uvec{p}_\perp &= \cos\phi\,\uvec{x} + \sin\phi\,\uvec{y}.
\end{align}

en la base circular definida por

\begin{equation}
\begin{aligned}
  \ket L &= \frac{1}{\sqrt 2} \left\{  \uvec x + i \uvec y \right\} \\
  \ket R &= \frac{1}{\sqrt 2} \left\{  \uvec x - i \uvec y \right\},
  \label{eq:def_of_circular_basis}
\end{aligned}
\end{equation}


las proyecciones perpendiculares de la base de fresnel se expresan como 


\begin{align}
\uvec{s}_\perp &=
\frac{i}{\sqrt{2}}\left(
e^{i\phi}\ket{L} - e^{-i\phi}\ket{R}
\right), \\
\uvec{p}_\perp &=
\frac{1}{\sqrt{2}}\left(
e^{-i\phi}\ket{L} + e^{i\phi}\ket{R}
\right).
\end{align}

Por tanto,

\begin{equation}
\vb{P}_\perp =
\frac{1}{\sqrt{2}}
e^{ik\left(
\frac{n r'^2}{2f} + (1-n)z(r')
\right)}
\Big\{ 
(t_p E_p + i t_s E_\phi)e^{i\phi}\ket{L} 
+
(t_p E_p - i t_s E_\phi)e^{-i\phi}\ket{R}
\Big\}
\end{equation}

y el campo queda




\begin{equation}
\begin{aligned}
  \mathbf E'(\mathbf r,f)=
  n\frac{e^{inkf}}{i\lambda f}
  e^{i\frac{nk}{2f}\|\mathbf r\|^2}
  \frac{1}{\sqrt{2}}
  \Bigg[
  &
  \mathcal F\left\{
  e^{ik\left(
  \frac{n r'^2}{2f} + (1-n)z(r')
  \right)}
  (t_pE_p+i t_sE_\phi)e^{i\phi}
  \right\}
  \left(\frac{n\mathbf r}{\lambda f}\right)\ket L
  \\
  +
  &
  \mathcal F\left\{
  e^{ik\left(
  \frac{n r'^2}{2f} + (1-n)z(r')
  \right)}
  (t_pE_p-i t_sE_\phi)e^{-i\phi}
  \right\}
  \left(\frac{n\mathbf r}{\lambda f}\right)\ket R

Para el termino $\ell_0$ dividiendo en la amplitud, lo aproximamos simplemente a $z_0$, para la fase la aproximación a de ser más cuidadosa, en concreto, partiendo de

    \ell_0 = \sqrt{x^2 + y^2 + (z(r) - z_0)^2} = \sqrt{r^2 + (z(r) - z_0)^2},

extrayendo el término $(z(r) - z_0)$ para realizar una expansión binomial en el régimen paraxial ($r \ll |z(r)-z_0|$):

\end{align}

\end{align}

\begin{equation}

\end{equation}

\end{align}

Así,

\begin{equation}
\mathscr F\{T_R\}(q,\varphi)
=
-
2\pi e^{-2i\varphi}
\int_0^\infty
(t_p-t_s)E_L
J_2(qr')r'\,dr'
+
2\pi
\int_0^\infty
(t_p+t_s)E_R
J_0(qr')r'\,dr'
\end{equation}


Resumiendo,


\begin{equation}
\mathscr F\{T_L\}(q,\varphi)
=
2\pi
\left[
\int_0^\infty
(t_p+t_s)E_L\,
J_0(qr')\,r'\,dr'
-
e^{2i\varphi}
\int_0^\infty
(t_p-t_s)E_R\,
J_2(qr')\,r'\,dr'
\right],
\label{eq:FTL_ts_tp}
\end{equation}

\begin{equation}
\mathscr F\{T_R\}(q,\varphi)
=
2\pi
\left[
-
e^{-2i\varphi}
\int_0^\infty
(t_p-t_s)E_L\,
J_2(qr')\,r'\,dr'
+
\int_0^\infty
(t_p+t_s)E_R\,
J_0(qr')\,r'\,dr'
\right].
\label{eq:FTR_ts_tp}
\end{equation}


Acá observamos las componentes $\mathscr F \{ T_L\}$ y $\mathscr F \{ T_R\}$ se pueden dividir en dos integrales, 
una que tiene $J_0$ y la otra con $J_2$ multiplicando los respectivos integrandos. Dado que las funciones la 
unica de las funciones de Bessel $J_m$ que tiene una contribución central (valor distinto de cero en $r=0$) y las demas 
solo tienen contribuciones laterales que son cada vez más alejadas del centro según el orden aumenta, como es ampliamente conocido en la teoría de las funciones de Bessel \cite{Watson1944}, y como se evidencia rapidamente en la figura \ref{fig:bessel_first_five}

El termino lateral, aumenta sus contribución en cuento mayor sea la diferencia entre $t_p$ y $t_s$, y el termino central
en cuanto mayor es la suma entre $t_p$ y $t_s$.


\begin{figure}[H]
\centering
\includegraphics[width=0.85\textwidth]{figures/bessel_first_five.png}
\caption{Primeras cinco funciones de Bessel de primera especie.}
\label{fig:bessel_first_five}
\end{figure}




En forma compacta, definiendo la transformada de Hankel de orden $m$ como 

\begin{equation}
\mathscr H_m[f](q)
=
\int_0^\infty
f(r')J_m(qr')r'\,dr',
\label{eq:def_Hankel_Transform_m_order}
\end{equation}


las ecuaciones \eqref{eq:FTL_ts_tp} y \eqref{eq:FTR_ts_tp} quedan expresadas como


\begin{equation}
\mathscr F\{T_L\}
=
2\pi \mathscr H_0[(t_p+t_s)E_L]
-
2\pi e^{2i\varphi}
\mathscr H_2[(t_p-t_s)E_R],
\end{equation}

y

\begin{equation}
\mathscr F\{T_R\}
=
-
2\pi e^{-2i\varphi}
\mathscr H_2[(t_p-t_s)E_L]
+
2\pi
\mathscr H_0[(t_p+t_s)E_R].
\end{equation}

Para conectar directamente con el desarrollo en series del apéndice \ref{apx:ovoids_series},
usamos explícitamente (cf. \eqref{eq:tp_plus_ts_series_paraxial} y \eqref{eq:tp_minus_ts_paraxial})
\begin{align*}
(t_p+t_s)(\rho') &= -\frac{n_0(n_0+n_i)^2(z_0-z_i)^2}{z_0^2z_i^2(n_0-n_i)^3}\,\rho'^2 + O(\rho'^4),\\[4pt]
(t_p-t_s)(\rho') &= \frac{4n_0}{n_i-n_0} -\frac{n_0(n_0+n_i)(z_0-z_i)^2}{z_0^2z_i^2(n_0-n_i)^2}\,\rho'^2 + O(\rho'^4),
\end{align*}
y luego sustituimos en las expresiones anteriores reteniendo términos hasta orden $r'^2$
con $\rho'^2=r'^2+O(r'^4)$. Con ello:

\begin{equation}
\begin{aligned}
\mathscr F\{T_L\}
\simeq{}&
-
2\pi
\frac{
n_0(n_0+n_i)^2(z_0-z_i)^2
}
{
z_0^2z_i^2(n_0-n_i)^3
}
\mathscr H_0[r'^2E_L]
\\[6pt]
&-
2\pi e^{2i\varphi}\!\left[
\frac{4 n_0}{n_i-n_0}\,\mathscr H_2[E_R]
-
\frac{
n_0(n_0+n_i)(z_0-z_i)^2
}
{
z_0^2z_i^2(n_0-n_i)^2
}
\mathscr H_2[r'^2E_R]
\right]
\end{aligned}
\label{eq:FT_TL_paraxial_main}
\end{equation}

\begin{equation}
\begin{aligned}
\mathscr F\{T_R\}
\simeq{}&
-
2\pi e^{-2i\varphi}\!\left[
\frac{4 n_0}{n_i-n_0}\,\mathscr H_2[E_L]
-
\frac{
n_0(n_0+n_i)(z_0-z_i)^2
}
{
z_0^2z_i^2(n_0-n_i)^2
}
\mathscr H_2[r'^2E_L]
\right]
\\[6pt]
&-
2\pi
\frac{
n_0(n_0+n_i)^2(z_0-z_i)^2
}
{
z_0^2z_i^2(n_0-n_i)^3
}
\mathscr H_0[r'^2E_R]
\end{aligned}
\label{eq:FT_TR_paraxial_main}
\end{equation}

Las ecuaciones \eqref{eq:FT_TL_paraxial_main} y
\eqref{eq:FT_TR_paraxial_main} son la forma paraxial, escrita solo con $(n_0,n_i,z_0,z_i)$ y consistente con la
convención de signos fijada por \eqref{eq:FTL_ts_tp},
\eqref{eq:FTR_ts_tp}, \eqref{eq:tp_plus_ts_series_paraxial} y
\eqref{eq:tp_minus_ts_paraxial}.




\subsection{Campo en el plano focal cuando el frente de onda inicial tiene amplitud homogénea}

Para hacer explícito el contenido físico de las ecuaciones
\eqref{eq:FT_TL_paraxial_main} y \eqref{eq:FT_TR_paraxial_main},
consideremos ahora el caso en que el frente de onda incidente tiene
amplitud homogénea sobre una pupila circular de radio \(R\). Es decir,
tomamos

\begin{equation}
E_L(r') = E_L\,\chi_R(r'),
\qquad
E_R(r') = E_R\,\chi_R(r'),
\end{equation}

donde \(E_L\) y \(E_R\) son constantes, y

\begin{equation}
\chi_R(r')
=
\begin{cases}
1, & 0\le r'\le R,\\
0, & r'>R.
\end{cases}
\end{equation}

Bajo esta hipótesis, las transformadas de Hankel que aparecen en
\eqref{eq:FT_TL_paraxial_main} y \eqref{eq:FT_TR_paraxial_main} pueden
evaluarse explícitamente. Para \(\sigma=L,R\), se obtiene

\begin{equation}
\mathscr H_0[r'^2E_\sigma](q)
=
E_\sigma
\left[
\frac{R^3}{q}J_1(qR)
+
\frac{2R^2}{q^2}J_0(qR)
-
\frac{4R}{q^3}J_1(qR)
\right],
\label{eq:H0_r2E_homogeneous_pupil}
\end{equation}

mientras que

\begin{equation}
\mathscr H_2[E_\sigma](q)
=
E_\sigma
\frac{
2\left[1-J_0(qR)\right]-qR J_1(qR)
}{q^2},
\label{eq:H2_E_homogeneous_pupil}
\end{equation}

y

\begin{equation}
\mathscr H_2[r'^2E_\sigma](q)
=
E_\sigma
\frac{R^3}{q}J_3(qR).
\label{eq:H2_r2E_homogeneous_pupil}
\end{equation}

Por tanto, sustituyendo
\eqref{eq:H0_r2E_homogeneous_pupil},
\eqref{eq:H2_E_homogeneous_pupil} y
\eqref{eq:H2_r2E_homogeneous_pupil} en
\eqref{eq:FT_TL_paraxial_main}, se tiene

\begin{equation}
\begin{aligned}
\mathscr F\{T_L\}
\simeq{}&
-
2\pi
\frac{
n_0(n_0+n_i)^2(z_0-z_i)^2
}
{
z_0^2z_i^2(n_0-n_i)^3
}
E_L
\left[
\frac{R^3}{q}J_1(qR)
+
\frac{2R^2}{q^2}J_0(qR)
-
\frac{4R}{q^3}J_1(qR)
\right]
\\[6pt]
&-
2\pi e^{2i\varphi}E_R
\left[
\frac{4n_0}{n_i-n_0}
\frac{
2\left[1-J_0(qR)\right]-qR J_1(qR)
}{q^2}
-
\frac{
n_0(n_0+n_i)(z_0-z_i)^2
}
{
z_0^2z_i^2(n_0-n_i)^2
}
\frac{R^3}{q}J_3(qR)
\right].
\end{aligned}
\label{eq:FT_TL_homogeneous_pupil}
\end{equation}

Análogamente, para la componente derecha,

\begin{equation}
\begin{aligned}
\mathscr F\{T_R\}
\simeq{}&
-
2\pi e^{-2i\varphi}E_L
\left[
\frac{4n_0}{n_i-n_0}
\frac{
2\left[1-J_0(qR)\right]-qR J_1(qR)
}{q^2}
-
\frac{
n_0(n_0+n_i)(z_0-z_i)^2
}
{
z_0^2z_i^2(n_0-n_i)^2
}
\frac{R^3}{q}J_3(qR)
\right]
\\[6pt]
&-
2\pi
\frac{
n_0(n_0+n_i)^2(z_0-z_i)^2
}
{
z_0^2z_i^2(n_0-n_i)^3
}
E_R
\left[
\frac{R^3}{q}J_1(qR)
+
\frac{2R^2}{q^2}J_0(qR)
-
\frac{4R}{q^3}J_1(qR)
\right].
\end{aligned}
\label{eq:FT_TR_homogeneous_pupil}
\end{equation}

Las expresiones anteriores deben entenderse por continuidad en \(q=0\).
En particular,

\begin{equation}
\mathscr H_0
=
\frac{E_\sigma R^4}{4},
\qquad
\mathscr H_2=0,
\qquad
\mathscr H_2=0.
\end{equation}

Las ecuaciones \eqref{eq:FT_TL_homogeneous_pupil} y
\eqref{eq:FT_TR_homogeneous_pupil} muestran que, incluso cuando el
campo incidente tiene amplitud homogénea en la pupila, el campo
transmitido no queda descrito únicamente por una transmisión escalar
constante. Los coeficientes de Fresnel introducen pesos radiales
distintos para las componentes \(s\) y \(p\), y dichos pesos generan
contenido espectral adicional en el plano focal.

Los términos diagonales, proporcionales a \(E_L\) en
\(\mathscr F\{T_L\}\) y a \(E_R\) en \(\mathscr F\{T_R\}\), preservan la
componente circular incidente. En cambio, los términos proporcionales a

\[
e^{2i\varphi}E_R,
\qquad
e^{-2i\varphi}E_L,
\]

mezclan las componentes circulares izquierda y derecha. Esta mezcla está
asociada a la diferencia \(t_p-t_s\), y por tanto desaparece solo cuando
las dos polarizaciones de Fresnel son transmitidas de manera idéntica.

Así, aun cuando la fase geométrica haya sido compensada por la condición
estigmática del dioptrio, la estructura vectorial de la refracción deja
una huella en la amplitud transmitida. Esta es precisamente una
aberración de amplitud: no proviene de una diferencia de camino óptico,
sino de la dependencia angular y polarizacional de los coeficientes de
Fresnel sobre la superficie refractante.



\begin{equation}
\begin{aligned}
\mathscr F\{T_L\}
\simeq
\frac{2\pi n_0}{(n_0-n_i)q^2}
\Bigg\{
&
8e^{2i\varphi}E_R
\\
&-
\left[
2
\frac{
(n_0+n_i)^2(z_0-z_i)^2R^2
}{
z_0^2z_i^2(n_0-n_i)^2
}
E_L
+
8e^{2i\varphi}E_R
\right]
J_0(qR)
\\
&+
\left[
\frac{
(n_0+n_i)^2(z_0-z_i)^2R(4-q^2R^2)
}{
z_0^2z_i^2(n_0-n_i)^2q
}
E_L
-
4Rq\,e^{2i\varphi}E_R
\right]
J_1(qR)
\\
&+
\left[
\frac{
(n_0+n_i)(z_0-z_i)^2R^3q
}{
z_0^2z_i^2(n_0-n_i)
}
e^{2i\varphi}E_R
\right]
J_3(qR)
\Bigg\}.
\end{aligned}
\label{eq:FT_TL_homogeneous_pupil_grouped}
\end{equation}

\begin{equation}
\begin{aligned}
\mathscr F\{T_R\}
\simeq
\frac{2\pi n_0}{(n_0-n_i)q^2}
\Bigg\{
&
8e^{-2i\varphi}E_L
\\
&-
\left[
8e^{-2i\varphi}E_L
+
2
\frac{
(n_0+n_i)^2(z_0-z_i)^2R^2
}{
z_0^2z_i^2(n_0-n_i)^2
}
E_R
\right]
J_0(qR)
\\
&+
\left[
-4Rq\,e^{-2i\varphi}E_L
+
\frac{
(n_0+n_i)^2(z_0-z_i)^2R(4-q^2R^2)
}{
z_0^2z_i^2(n_0-n_i)^2q
}
E_R
\right]
J_1(qR)
\\
&+
\left[
\frac{
(n_0+n_i)(z_0-z_i)^2R^3q
}{
z_0^2z_i^2(n_0-n_i)
}
e^{-2i\varphi}E_L
\right]
J_3(qR)
\Bigg\}.
\end{aligned}
\label{eq:FT_TR_homogeneous_pupil_grouped}
\end{equation}
